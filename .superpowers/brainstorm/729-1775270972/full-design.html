一、数据库设计

CREATE TABLE env_machine_log (
    id UUID PRIMARY KEY,
    namespace VARCHAR(64) NOT NULL,       -- 机器分类
    machine_id UUID NOT NULL,              -- 机器ID
    ip VARCHAR(45),                        -- 机器IP
    device_type VARCHAR(20) NOT NULL,      -- 机器类型
    device_sn VARCHAR(64),                 -- 设备SN
    mark VARCHAR(255),                     -- 申请的标签（用于TOP10统计）
    action VARCHAR(20) NOT NULL,           -- apply/release
    result VARCHAR(20) NOT NULL,           -- success/fail
    fail_reason VARCHAR(255),              -- 失败原因（env not enough）
    apply_time TIMESTAMP,                  -- 申请时间
    release_time TIMESTAMP,                 -- 释放时间
    duration_minutes INTEGER,              -- 占用时长（分钟）
    sys_create_datetime TIMESTAMP,

    INDEX (sys_create_datetime),           -- 用于清理和统计
    INDEX (namespace, sys_create_datetime),
    INDEX (machine_id)
);

二、数据清理任务

任务函数（添加到 tasks.py）
async def cleanup_env_machine_log_task(job_code: str = None, days: int = 7):
    """清理执行机申请日志，保留7天"""
    cutoff = datetime.now() - timedelta(days=days)
    await db.execute(delete(EnvMachineLog).where(
        EnvMachineLog.sys_create_datetime < cutoff
    ))
